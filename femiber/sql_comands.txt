UPDATE cases
SET consang_kinship = processed_kinship
FROM (
  SELECT REPLACE(REPLACE(consang_kinship, ' ', ', '), '*', '') AS processed_kinship
  FROM cases
) AS col;

ALTER TABLE cases
DROP COLUMN problems;

UPDATE people
SET people_filter = 
    CASE 
        WHEN standard_name LIKE 'Anonymae%' THEN 'groups'
        WHEN standard_name LIKE 'Anonyma%' THEN 'individual'
        ELSE SUBSTRING(standard_name, 1, 1)
    END;
